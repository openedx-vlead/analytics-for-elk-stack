* How to run script
  1) Clone the repository
     #+BEGIN_EXAMPLE
     git clone https://github.com/vlead/analytics-db
     #+END_EXAMPLE
  2) Change directory to =analytics-db= and checkout branch
     to =develop=
     #+BEGIN_EXAMPLE
     cd analytics-db
     git checkout develop
     #+END_EXAMPLE
  3) Build the sources
     #+BEGIN_EXAMPLE
     make
     #+END_EXAMPLE
  4) Change directory to =deployment=
     #+BEGIN_EXAMPLE
     cd build/code/deployment
     #+END_EXAMPLE
  5) Configure the =analytics_host= url in =map-labids.py=
     #+BEGIN_EXAMPLE
     ----
     ----
     analytics_host = "http://192.168.33.4:9200"
     ----
     ----
     #+END_EXAMPLE
  6) Run the script
     #+BEGIN_EXAMPLE
     python map-labids.py

** Get analytics records
   Get the analytics records for a given lab_id
   #+NAME: get-analytics-record
   #+BEGIN_SRC python

def get_analytics_records(lab_id):

    analytics_url = "%s/vlabs/usage/_search?q=LAB_ID:%s&size=10000" % (analytics_host, lab_id)
    try:
        r = requests.get(analytics_url)
        if r.status_code == 200:
            analytics_records = r.json()['hits']['hits']
            for record in analytics_records:
                if str(record['_source']['LAB_ID']) == lab_id:
                    update_analytics_record(record)
        else:
            print "Error occured"
    except Exception as e:
        print str(e)

   #+END_SRC
** Update analytics record
   This function updates analytics records
   #+NAME: put-analytics-record
   #+BEGIN_SRC python

def update_analytics_record(record):

    old_lab_id = str(record['_source']['LAB_ID']).lower()
    record_id = str(record['_id'])
    new_lab_id = get_new_lab_id(old_lab_id)
    analytics_url = "%s/vlabs/usage/%s" % (analytics_host, record_id)
    record['_source']['LAB_ID'] = new_lab_id
    headers = {'Content-Type': 'application/json'}
    r = requests.put(analytics_url, data=json.dumps(record['_source']), headers=headers)

    if r.status_code == 201 or r.status_code == 200:
        print "updated lab_id from %s ---> %s" % (old_lab_id, new_lab_id)
    else:
        print "error in updating lab_id from %s ---> %s" % (old_lab_id, new_lab_id)

   #+END_SRC
** Get new lab id
   This function maps the old lab_id and returns the new
   lab_id
   #+NAME: get-new-labid
   #+BEGIN_SRC python

def get_new_lab_id(lab_id):
    try:
        with open("lab-ids-mapping.json") as json_data:
            labs = json.load(json_data)
        for lab in labs:
            if lab_id.lower() == str(lab['old_lab_id']).lower():
                return lab['new_lab_id']
    except Exception as e:
        print str(e)

   #+END_SRC
** Map lab ids
   #+NAME: map-lab-ids
   #+BEGIN_SRC python

def map_lab_ids():
    try:
        with open("lab-ids-mapping.json") as json_data:
            labs = json.load(json_data)
        for lab in labs:
            lab_id = str(lab['old_lab_id']).upper()
            get_analytics_records(lab_id)
    except Exception as e:
        print str(e)
